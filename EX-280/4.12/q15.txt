oc new-app nginx --name=app -n project4

apiVersion: apps/v1
kind: Deployment
metadata:
  name: app
  namespace: project4
spec:
  replicas: 1
  selector:
    matchLabels:
      app: app
  template:
    metadata:
      labels:
        app: app
    spec:
      containers:
      - name: app
        image: nginx
        ports:
        - containerPort: 80

oc apply -f project4-app.yaml
oc expose deployment app --port=80 -n project4
oc get svc -n project4

oc expose svc app -n project4
oc get route -n project4
